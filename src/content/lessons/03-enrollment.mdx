## How progress tracking works

After enrollment, lessons can be marked as complete in two ways:

- **From the website** — use the checkbox at the bottom of any lesson page on opencode.school.
- **From OpenCode** — as you work through lessons, OpenCode can call the progress API to mark lessons complete on your behalf.

Your progress syncs between the website and your OpenCode app, so you can pick up where you left off from either one.

## Already have a student ID?

If you enrolled from OpenCode and were given a link like `https://opencode.school?sid=curious-hacker-2019`, just visit that URL and your progress will be linked automatically.

## What just happened?

You just used OpenCode for the first time! You typed a [prompt](/glossary#prompt) — a plain-English instruction — and OpenCode acted on it. That's the core of how OpenCode works: you describe what you want, and the [agent](/glossary#agent) figures out how to do it.

In the next lesson, we'll set up a configuration file that tells OpenCode how to behave across all your projects.

<script is:inline>{`
(function() {
  var enrollBtn = document.getElementById("enroll-button");
  var enrollError = document.getElementById("enroll-error");
  var notEnrolled = document.getElementById("not-enrolled");
  var enrolled = document.getElementById("enrolled");
  var studentIdDisplay = document.getElementById("student-id-display");
  var promptStudentId = document.getElementById("prompt-student-id");
  var copyBtn = document.getElementById("copy-prompt");

  if (!enrollBtn || !window.school) return;

  function showEnrolled(studentId) {
    notEnrolled.classList.add("hidden");
    enrolled.classList.remove("hidden");
    studentIdDisplay.textContent = studentId;
    promptStudentId.textContent = studentId;
  }

  // Check if already enrolled
  var existingId = window.school.getStudentId();
  if (existingId) {
    showEnrolled(existingId);
  }

  enrollBtn.addEventListener("click", async function() {
    enrollBtn.disabled = true;
    enrollBtn.textContent = "Enrolling...";
    enrollError.classList.add("hidden");

    try {
      var data = await window.school.enroll();
      showEnrolled(data.studentId);

      // Show the mark-complete section on this page too
      var completeSection = document.getElementById("lesson-complete-section");
      if (completeSection) completeSection.classList.remove("hidden");
    } catch (e) {
      enrollError.classList.remove("hidden");
      enrollBtn.disabled = false;
      enrollBtn.textContent = "Enroll in OpenCode School";
    }
  });

  if (copyBtn) {
    copyBtn.addEventListener("click", function() {
      var prompt = document.querySelector("#opencode-prompt p");
      if (prompt) {
        navigator.clipboard.writeText(prompt.textContent.trim());
        copyBtn.textContent = "copied!";
        setTimeout(function() { copyBtn.textContent = "copy"; }, 2000);
      }
    });
  }
})();
`}</script>
