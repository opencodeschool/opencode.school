## A computer

OpenCode runs on **macOS**, **Windows**, and **Linux**. Any computer from the last 5-6 years should work fine. OpenCode lets you run AI models hosted in the cloud, so you don't need a powerful machine — just something that can run a web browser and connect to the internet.

You can also configure OpenCode to use models running locally on your own machine using tools like [Ollama](https://ollama.com) and [LM Studio](https://lmstudio.ai), but that's optional and not required for completing the OpenCode School curriculum.

## That's it!

No special software to install yet, no accounts to create. If you have a computer with an internet connection, you're ready for the next lesson.

<script>{`
document.addEventListener("DOMContentLoaded", function() {
  var CHECK = '<span class="theme-solid-bg inline-flex w-5 h-5 rounded-full text-white text-[11px] font-bold items-center justify-center">\\u2713</span>';
  var WARN = '<span class="text-yellow-500 dark:text-yellow-400">!</span>';
  var DASH = '<span class="text-gray-300 dark:text-stone-600">\\u2014</span>';

  function setCheck(id, value, icon) {
    var row = document.getElementById(id);
    if (!row) return;
    var val = row.querySelector(".check-value");
    var ico = row.querySelector(".check-icon");
    if (val) val.textContent = value;
    if (ico) ico.innerHTML = icon;
  }

  function delay(ms) {
    return new Promise(function(r) { setTimeout(r, ms); });
  }

  // Detect OS from user agent
  function detectOS() {
    var ua = navigator.userAgent;
    if (/Macintosh|Mac OS X/.test(ua)) return "macOS";
    if (/Windows/.test(ua)) return "Windows";
    if (/Linux/.test(ua)) return "Linux";
    if (/CrOS/.test(ua)) return "ChromeOS";
    return "Unknown";
  }

  // Measure connectivity: try navigator.connection first, fall back to fetch timing
  async function measureConnectivity() {
    // Try Network Information API (Chromium only)
    var conn = navigator.connection || navigator.mozConnection || navigator.webkitConnection;
    if (conn && conn.rtt !== undefined && conn.rtt > 0) {
      var rtt = conn.rtt;
      var label = "Connected, ~" + rtt + "ms latency";
      var icon = rtt < 300 ? CHECK : WARN;
      return { label: label, icon: icon };
    }

    // Fallback: time a small fetch
    if (!navigator.onLine) {
      return { label: "Offline", icon: WARN };
    }

    try {
      var start = performance.now();
      await fetch("/api/enroll", { method: "HEAD", cache: "no-store" }).catch(function() {
        // HEAD might 405, that's fine — we only need the round trip
      });
      var elapsed = Math.round(performance.now() - start);
      var label2 = "Connected, ~" + elapsed + "ms latency";
      var icon2 = elapsed < 500 ? CHECK : WARN;
      return { label: label2, icon: icon2 };
    } catch (e) {
      return { label: navigator.onLine ? "Connected" : "Offline", icon: navigator.onLine ? CHECK : WARN };
    }
  }

  // Run all checks with staggered reveal
  async function runChecks() {
    var allPassed = true;

    // OS
    await delay(300);
    var os = detectOS();
    var supportedOS = ["macOS", "Windows", "Linux"];
    var osIcon = supportedOS.indexOf(os) !== -1 ? CHECK : WARN;
    setCheck("check-os", os, osIcon);
    if (osIcon !== CHECK) allPassed = false;

    // CPU
    await delay(400);
    var cores = navigator.hardwareConcurrency;
    var cpuIcon;
    if (cores) {
      cpuIcon = cores >= 2 ? CHECK : WARN;
      setCheck("check-cpu", "\u2265" + cores + " cores", cpuIcon);
    } else {
      cpuIcon = DASH;
      setCheck("check-cpu", "Unable to detect", DASH);
    }
    if (cpuIcon !== CHECK) allPassed = false;

    // Memory
    await delay(400);
    var mem = navigator.deviceMemory;
    var memIcon;
    if (mem) {
      memIcon = mem >= 4 ? CHECK : WARN;
      setCheck("check-memory", mem >= 1 ? "\u2265" + mem + " GB" : "\u2265" + (mem * 1024) + " MB", memIcon);
    } else {
      memIcon = DASH;
      setCheck("check-memory", "Unable to detect", DASH);
    }
    if (memIcon !== CHECK) allPassed = false;

    // Internet
    await delay(300);
    var result = await measureConnectivity();
    setCheck("check-internet", result.label, result.icon);
    if (result.icon !== CHECK) allPassed = false;

    // Reveal inline next button if all checks passed
    if (allPassed) {
      var nextEl = document.getElementById("system-check-next");
      var nextLink = document.getElementById("system-check-next-link");
      var nextLabel = document.getElementById("system-check-next-label");
      if (nextEl && nextLink && nextLabel) {
        var enrolled = window.school && window.school.getStudentId && window.school.getStudentId();
        if (!enrolled) {
          nextLink.setAttribute("href", "/");
          nextLabel.textContent = "Enroll for free to get started \u2192";
        } else {
          // Intercept click: mark lesson complete, then navigate
          nextLink.addEventListener("click", function(e) {
            e.preventDefault();
            var dest = nextLink.getAttribute("href");
            nextLabel.textContent = "Saving\u2026";
            nextLink.style.pointerEvents = "none";
            window.school.markComplete("prerequisites").then(function(progress) {
              if (progress) window.school.updateAllCheckmarks();
              window.location.href = dest;
            });
          });
        }
        nextEl.classList.remove("hidden");
      }
    }
  }

  runChecks();
});
`}</script>
