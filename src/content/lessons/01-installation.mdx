In this course we'll be using **OpenCode Desktop** — a native app with a [GUI](/glossary#gui) (graphical user interface) you interact with using your mouse and keyboard. If you've used apps like ChatGPT or Claude, this chat-based interface should feel pretty familiar to you.

OpenCode also has a [TUI](https://opencode.ai/docs/tui/) (terminal user interface) for terminal-based workflows, a [CLI](https://opencode.ai/docs/cli/) (command-line interface) for one-off commands, and [IDE extensions](https://opencode.ai/docs/ide/) for VS Code, Cursor, Zed, and Windsurf. The skills you learn here apply to any of these interfaces. But for this course, **Desktop is all you need**.

Before we install anything, let's run a quick system check. Your browser can detect most of this automatically:

## Launch the app

Double-click OpenCode Desktop from your applications menu or dock, or use Spotlight to open it.

## Open a project

You'll be prompted to open a project — click **Open Project** and select any folder on your computer. You should then see a chat interface with a text input at the bottom.

## Enter a prompt

Try this prompt to confirm everything is working:

> Hello, are you there?

<span id="send-key-hint">Press <strong>Return</strong> to send.</span> If you get a response, you're connected to a language model and ready to go.

<script>{`
document.addEventListener("DOMContentLoaded", function() {
  var CHECK = '<span class="theme-solid-bg inline-flex w-5 h-5 rounded-full text-white text-[11px] font-bold items-center justify-center">\\u2713</span>';
  var WARN = '<span class="text-yellow-500 dark:text-yellow-400">!</span>';
  var DASH = '<span class="text-gray-300 dark:text-stone-600">\\u2014</span>';

  function setCheck(id, value, icon) {
    var row = document.getElementById(id);
    if (!row) return;
    var val = row.querySelector(".check-value");
    var ico = row.querySelector(".check-icon");
    if (val) val.textContent = value;
    if (ico) ico.innerHTML = icon;
  }

  function delay(ms) {
    return new Promise(function(r) { setTimeout(r, ms); });
  }

  // Detect OS from user agent
  function detectOS() {
    var ua = navigator.userAgent;
    if (/Macintosh|Mac OS X/.test(ua)) return "macOS";
    if (/Windows/.test(ua)) return "Windows";
    if (/Linux/.test(ua)) return "Linux";
    if (/CrOS/.test(ua)) return "ChromeOS";
    return "Unknown";
  }

  // Measure connectivity: try navigator.connection first, fall back to fetch timing
  async function measureConnectivity() {
    var conn = navigator.connection || navigator.mozConnection || navigator.webkitConnection;
    if (conn && conn.rtt !== undefined && conn.rtt > 0) {
      var rtt = conn.rtt;
      var label = "Connected, ~" + rtt + "ms latency";
      var icon = rtt < 300 ? CHECK : WARN;
      return { label: label, icon: icon };
    }

    if (!navigator.onLine) {
      return { label: "Offline", icon: WARN };
    }

    try {
      var start = performance.now();
      await fetch("/api/enroll", { method: "HEAD", cache: "no-store" }).catch(function() {});
      var elapsed = Math.round(performance.now() - start);
      var label2 = "Connected, ~" + elapsed + "ms latency";
      var icon2 = elapsed < 500 ? CHECK : WARN;
      return { label: label2, icon: icon2 };
    } catch (e) {
      return { label: navigator.onLine ? "Connected" : "Offline", icon: navigator.onLine ? CHECK : WARN };
    }
  }

  // Run system checks with staggered reveal
  async function runChecks() {
    await delay(300);
    var os = detectOS();
    var supportedOS = ["macOS", "Windows", "Linux"];
    var osIcon = supportedOS.indexOf(os) !== -1 ? CHECK : WARN;
    setCheck("check-os", os, osIcon);

    await delay(400);
    var cores = navigator.hardwareConcurrency;
    if (cores) {
      setCheck("check-cpu", "\\u2265" + cores + " cores", cores >= 2 ? CHECK : WARN);
    } else {
      setCheck("check-cpu", "Unable to detect", DASH);
    }

    await delay(400);
    var mem = navigator.deviceMemory;
    if (mem) {
      setCheck("check-memory", mem >= 1 ? "\\u2265" + mem + " GB" : "\\u2265" + (mem * 1024) + " MB", mem >= 4 ? CHECK : WARN);
    } else {
      setCheck("check-memory", "Unable to detect", DASH);
    }

    await delay(300);
    var result = await measureConnectivity();
    setCheck("check-internet", result.label, result.icon);

    var allPassed =
      ["check-os", "check-cpu", "check-memory", "check-internet"].every(function(id) {
        var row = document.getElementById(id);
        return row && row.querySelector(".check-icon").innerHTML.indexOf("theme-solid-bg") !== -1;
      });
    if (allPassed) {
      var successEl = document.getElementById("system-check-success");
      if (successEl) successEl.classList.remove("hidden");
    }
  }

  // OS detection for install section and send-key hint
  var ua = navigator.userAgent;
  var os = "other";
  if (/Macintosh|Mac OS X/.test(ua)) os = "macos";
  else if (/Windows/.test(ua)) os = "windows";
  else if (/Linux/.test(ua)) os = "linux";

  var loading = document.getElementById("install-loading");
  if (loading) loading.classList.add("hidden");

  var section = document.getElementById("install-" + os);
  if (section) section.classList.remove("hidden");

  var hint = document.getElementById("send-key-hint");
  if (hint) hint.innerHTML = os === "windows" ? "Press <strong>Enter</strong> to send." : "Press <strong>Return</strong> to send.";

  // Brew copy button
  var brewBtn = document.querySelector(".brew-copy");
  if (brewBtn) {
    brewBtn.addEventListener("click", function() {
      navigator.clipboard.writeText("brew install --cask opencode-desktop").then(function() {
        brewBtn.textContent = "Copied!";
        setTimeout(function() { brewBtn.textContent = "Copy"; }, 2000);
      });
    });
  }

  runChecks();
});
`}</script>
