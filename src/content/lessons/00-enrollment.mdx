<script>{`
document.addEventListener("DOMContentLoaded", function() {
  var enrollBtn = document.getElementById("enroll-button");
  var enrollError = document.getElementById("enroll-error");
  var notEnrolled = document.getElementById("not-enrolled");
  var enrolled = document.getElementById("enrolled");
  var display = document.getElementById("student-id-display");

  if (!enrollBtn || !window.school) return;

  var chars = "abcdefghijklmnopqrstuvwxyz";
  var digits = "0123456789";

  function randomChar(real) {
    if (real === "-") return "-";
    if (real >= "0" && real <= "9") return digits[Math.floor(Math.random() * digits.length)];
    return chars[Math.floor(Math.random() * chars.length)];
  }

  function scramble(target) {
    var result = "";
    for (var i = 0; i < target.length; i++) {
      result += randomChar(target[i]);
    }
    return result;
  }

  function fireConfetti() {
    var duration = 2000;
    var end = Date.now() + duration;
    var colors = ["#3b82f6", "#60a5fa", "#2563eb", "#818cf8", "#6366f1"];

    (function frame() {
      window.confetti({
        particleCount: 3,
        angle: 60,
        spread: 55,
        origin: { x: 0 },
        colors: colors
      });
      window.confetti({
        particleCount: 3,
        angle: 120,
        spread: 55,
        origin: { x: 1 },
        colors: colors
      });
      if (Date.now() < end) requestAnimationFrame(frame);
    })();
  }

  function cipherReveal(target, callback) {
    var resolved = 0;
    var current = scramble(target);
    display.textContent = current;

    var scrambleInterval = setInterval(function() {
      var next = "";
      for (var i = 0; i < target.length; i++) {
        if (i < resolved) {
          next += target[i];
        } else if (target[i] === "-") {
          next += "-";
        } else {
          next += randomChar(target[i]);
        }
      }
      current = next;
      display.textContent = current;
    }, 30);

    setTimeout(function() {
      var resolveInterval = setInterval(function() {
        if (resolved >= target.length) {
          clearInterval(scrambleInterval);
          clearInterval(resolveInterval);
          display.textContent = target;
          if (callback) callback();
          return;
        }
        if (target[resolved] === "-") {
          resolved++;
        }
        resolved++;
      }, 140);
    }, 600);
  }

  function showEnrolled(studentId, animate) {
    notEnrolled.classList.add("hidden");
    enrolled.classList.remove("hidden");

    // Hide the lesson description and header prose since we show our own enrolled message
    var article = document.querySelector("article.prose");
    if (article) {
      // Hide the intro paragraph that's part of the prose wrapper
      var proseChildren = article.querySelectorAll(":scope > p");
      proseChildren.forEach(function(p) {
        if (p.closest("#enrollment-area")) return;
        p.classList.add("hidden");
      });
    }

    if (animate) {
      cipherReveal(studentId, function() {
        fireConfetti();
        // Auto-complete the enrollment lesson
        window.school.markComplete("enrollment").then(function() {
          window.school.updateAllCheckmarks();
          // Update the enrolled-as indicator in the layout
          if (typeof window.school.updateEnrolledIndicator === "function") {
            window.school.updateEnrolledIndicator();
          }
        });
      });
    } else {
      display.textContent = studentId;
    }
  }

  var existingId = window.school.getStudentId();
  if (existingId) {
    showEnrolled(existingId, false);
  }

  enrollBtn.addEventListener("click", async function() {
    enrollBtn.disabled = true;
    enrollBtn.textContent = "Generating...";
    enrollError.classList.add("hidden");

    try {
      var data = await window.school.enroll();
      showEnrolled(data.studentId, true);
    } catch (e) {
      enrollError.classList.remove("hidden");
      enrollBtn.disabled = false;
      enrollBtn.textContent = "Generate my student ID";
    }
  });
});
`}</script>
