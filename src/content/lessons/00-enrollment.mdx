## Already have a student ID?

If you enrolled from OpenCode and were given a link like `https://opencode.school?sid=curious-hacker-2019`, just visit that URL and your progress will be linked automatically.

<script>{`
document.addEventListener("DOMContentLoaded", function() {
  var enrollBtn = document.getElementById("enroll-button");
  var enrollError = document.getElementById("enroll-error");
  var notEnrolled = document.getElementById("not-enrolled");
  var enrolled = document.getElementById("enrolled");
  var studentIdDisplay = document.getElementById("student-id-display");

  if (!enrollBtn || !window.school) return;

  function showEnrolled(studentId) {
    notEnrolled.classList.add("hidden");
    enrolled.classList.remove("hidden");
    studentIdDisplay.textContent = studentId;
  }

  var existingId = window.school.getStudentId();
  if (existingId) {
    showEnrolled(existingId);
  }

  enrollBtn.addEventListener("click", async function() {
    enrollBtn.disabled = true;
    enrollBtn.textContent = "Generating...";
    enrollError.classList.add("hidden");

    try {
      var data = await window.school.enroll();
      showEnrolled(data.studentId);

      var completeSection = document.getElementById("lesson-complete-section");
      if (completeSection) completeSection.classList.remove("hidden");
    } catch (e) {
      enrollError.classList.remove("hidden");
      enrollBtn.disabled = false;
      enrollBtn.textContent = "Generate my student ID";
    }
  });
});
`}</script>
